# Criação do database dos times
create database dbTimeFutebol;

# Ativa o database a ser utilizado
	use dbTimeFutebol;

create table tblTimes(
	idTimes int auto_increment not null primary key,
    nome varchar(100) not null

);

create table tblPosicoes(
	idPosicoes int auto_increment not null primary key,
    posicao varchar(100) not null

);

create table tblJogadores(
	idJogadores int auto_increment not null,
    nome varchar(100) not null,
    idPosicoes int not null,
    primary key (idJogadores),
	constraint FK_posicao_jogadores foreign key (idPosicoes) 
    references tblPosicoes (idPosicoes)
);
